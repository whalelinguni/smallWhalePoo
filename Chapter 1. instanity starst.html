powershell
# Define the GitHub repository URL
$repoUrl = "https://api.github.com/repos/f-droid/fdroidclient/releases/latest"

# Define the destination path for the APK file
$destinationPath = "C:\Path\To\Save\fdroidclient-latest.apk"

# Function to download the latest release APK
function Download-LatestFdroidClient {
    param (
        [string]$url,
        [string]$outputPath
    )

    try {
        # Use Invoke-RestMethod to get the latest release data
        $releaseData = Invoke-RestMethod -Uri $url -Headers @{ "User-Agent" = "PowerShell" }

        # Find the APK asset in the release assets
        $apkAsset = $releaseData.assets | Where-Object { $_.name -like "*.apk" }

        if ($apkAsset -ne $null) {
            # Download the APK file
            Write-Host "Downloading $($apkAsset.name) from $($apkAsset.browser_download_url)..."
            Invoke-WebRequest -Uri $apkAsset.browser_download_url -OutFile $outputPath
            Write-Host "Download complete. Saved to $outputPath"
        } else {
            Write-Host "No APK file found in the latest release."
        }
    } catch {
        Write-Error "An error occurred: $_"
    }
}

# Call the function to download the latest release APK
Download-LatestFdroidClient -url $repoUrl -outputPath $destinationPath
